    4. Add the handleActivate function:
       ```tsx
       const handleActivate = async () => {
         const key = keyInputValue.trim();
         if (!key) return;
         setIsActivating(true);
         setKeyError(null);
         try {
           const result = await Promise.race([
             fetch(`${WORKER_URL}/validate-key`, {
               method: 'POST',
               headers: { 'Content-Type': 'application/json' },
               body: JSON.stringify({ key }),
             }).then(r => r.json() as Promise<{ valid: boolean }>),
             new Promise<never>((_, reject) => setTimeout(() => reject(new Error('timeout')), 5000)),
           ]);
           if (result.valid) {
             await storageLocalLicenseKey.setValue(key);
             setIsPremium(true);
             setShowKeyInput(false);
             setKeyInputValue('');
           } else {
             setKeyError('Invalid key — please check and try again.');
           }
         } catch {
           setKeyError('Activation failed — check your connection and try again.');
         } finally {
           setIsActivating(false);
         }
       };
       ```
       WORKER_URL: use the same pattern already present in the file (`import.meta.env.VITE_WORKER_URL`). The `/validate-key` path is appended inline, not a separate constant.

       {!isPremium && showKeyInput && (
